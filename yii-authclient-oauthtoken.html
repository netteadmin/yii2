                protected function defaultExpireDurationParamKey()
{
    $expireDurationParamKey = 'expires_in';
    foreach ($this->getParams() as $name => $value) {
        if (strpos($name, 'expir') !== false) {
            $expireDurationParamKey = $name;
            break;
        }
    }
    return $expireDurationParamKey;
}
