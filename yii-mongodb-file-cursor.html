                #[\ReturnTypeWillChange]
public function current()
{
    $value = parent::current();
    if (!isset($value['file'])) {
        $value['file'] = $this->collection->createDownload(array_intersect_key($value, ['_id' => true, 'filename' => true, 'length' => true, 'chunkSize' => true]));
    }
    return $value;
}
